  <script>

    gsap.registerPlugin(ScrollTrigger);


    const sections = gsap.utils.toArray(".section");

    const dots = gsap.utils.toArray(".dot");


    // Initial state

    gsap.set(sections[0], { autoAlpha: 1, visibility: "visible" });

    gsap.set(dots[0], { opacity: 1, scale: 1.5 });


    const tl = gsap.timeline({

      scrollTrigger: {

        trigger: ".viewport",

        start: "top top",

        end: "bottom+=500% top",

        scrub: 1.5,

        pin: true,

        snap: 1 / (sections.length - 1)

      }

    });


    sections.forEach((section, i) => {

      if (i > 0) {

        // Zoom-in effect for the background of the PREVIOUS section

        tl.to(sections[i-1].querySelector(".bg-image"), {

          scale: 4,

          opacity: 0,

          duration: 1.5,

          ease: "power2.in"

        }, i)

        

        // Zoom-out effect for the text of the PREVIOUS section

        .to(sections[i-1].querySelector(".content"), {

          scale: 0.8,

          opacity: 0,

          duration: 1,

          ease: "power2.in"

        }, i)


        // Fade in CURRENT section

        .to(section, {

          autoAlpha: 1,

          visibility: "visible",

          duration: 1

        }, i + 0.5)


        // Update Nav Dots

        .to(dots[i-1], { opacity: 0.3, scale: 1 }, i + 0.5)

        .to(dots[i], { opacity: 1, scale: 1.5 }, i + 0.5);

      }

    });


    // Suble parallax movement on mouse

    window.addEventListener("mousemove", (e) => {

        const x = (e.clientX / window.innerWidth - 0.5) * 20;

        const y = (e.clientY / window.innerHeight - 0.5) * 20;

        gsap.to(".bg-image", { x: x, y: y, duration: 2, ease: "power2.out" });

    });

  </script>
